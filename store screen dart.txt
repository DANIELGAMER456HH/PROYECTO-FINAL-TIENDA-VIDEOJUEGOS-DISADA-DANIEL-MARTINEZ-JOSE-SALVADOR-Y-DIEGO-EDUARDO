import 'package:flutter/material.dart';
import 'package:game_vault/models/game.dart';
import 'package:game_vault/models/purchase.dart';
import 'package:game_vault/services/api_service.dart';
import 'package:game_vault/constants/app_constants.dart';
import 'package:uuid/uuid.dart';

/// Pantalla de Tienda de Videojuegos
///
/// Permite a los usuarios:
/// - Ver la lista de videojuegos disponibles
/// - Seleccionar juegos para comprar
/// - Llenar un formulario con sus datos personales
/// - Procesar la compra
class StoreScreen extends StatefulWidget {
  const StoreScreen({Key? key}) : super(key: key);

  @override
  State<StoreScreen> createState() => _StoreScreenState();
}

class _StoreScreenState extends State<StoreScreen> {
  /// Instancia del servicio de API
  final ApiService _apiService = ApiService();

  /// Lista de todos los juegos disponibles
  List<Game> _allGames = [];

  /// Juegos seleccionados para la compra (carrito)
  List<Game> _selectedGames = [];

  /// Estado de carga
  bool _isLoading = true;

  /// Mostrar vista del carrito
  bool _showCart = false;

  /// Mostrar formulario de compra
  bool _showCheckout = false;

  /// Controladores de formulario
  late TextEditingController _nameController;
  late TextEditingController _ageController;
  late TextEditingController _emailController;
  late TextEditingController _phoneController;
  late TextEditingController _cityController;

  /// Controlador de búsqueda
  late TextEditingController _searchController;

  /// Key del formulario para validación
  final _formKey = GlobalKey<FormState>();

  /// Booleano para mostrar mensaje de éxito
  bool _purchaseSuccess = false;

  @override
  void initState() {
    super.initState();
    _nameController = TextEditingController();
    _ageController = TextEditingController();
    _emailController = TextEditingController();
    _phoneController = TextEditingController();
    _cityController = TextEditingController();
    _searchController = TextEditingController();

    // Cargar juegos de la API
    _loadGames();
  }

  @override
  void dispose() {
    _nameController.dispose();
    _ageController.dispose();
    _emailController.dispose();
    _phoneController.dispose();
    _cityController.dispose();
    _searchController.dispose();
    super.dispose();
  }

  /// Carga los juegos desde la API
  Future<void> _loadGames() async {
    try {
      setState(() => _isLoading = true);

      final games = await _apiService.getPopularGames();

      setState(() {
        _allGames = games;
        _isLoading = false;
      });

      print(' ${games.length} juegos cargados para la tienda');
    } catch (e) {
      setState(() => _isLoading = false);

      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(
            content: Text(' Error al cargar juegos: $e'),
            backgroundColor: Colors.red,
          ),
        );
      }
    }
  }

  /// Agregar o remover juego del carrito
  void _toggleGameSelection(Game game) {
    setState(() {
      if (_selectedGames.any((g) => g.id == game.id)) {
        _selectedGames.removeWhere((g) => g.id == game.id);
        print(' ${game.name} removido del carrito');
      } else {
        _selectedGames.add(game);
        print(' ${game.name} agregado al carrito');
      }
    });
  }

  /// Procesar la compra
  void _processPurchase() {
    if (_formKey.currentState!.validate()) {
      if (_selectedGames.isEmpty) {
        ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(
            content: Text(' Selecciona al menos un juego'),
            backgroundColor: Colors.orange,
          ),
        );
        return;
      }

      // Crear la compra
      final purchase = Purchase(
        id: const Uuid().v4(),
        name: _nameController.text,
        age: int.parse(_ageController.text),
        email: _emailController.text,
        phone: _phoneController.text,
        city: _cityController.text,
        games: _selectedGames,
        purchaseDate: DateTime.now(),
        total: Purchase.calculateTotal(_selectedGames),
      );

      // Guardar la compra (simulado)
      print(' Guardando compra: ${purchase.id}');

      // Mostrar mensaje de éxito
      setState(() {
        _purchaseSuccess = true;
        _showCheckout = false;
      });

      // Limpiar
      _clearForm();
      _selectedGames.clear();

      // Mostrar snackbar
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
          content: Text(
            ' ¡Compra realizada con éxito!\n'
            'ID: ${purchase.id.substring(0, 8)}\n'
            'Total juegos: ${purchase.games.length}',
          ),
          backgroundColor: Colors.green,
          duration: const Duration(seconds: 4),
        ),
      );

      // Resetear después de 2 segundos
      Future.delayed(const Duration(seconds: 2), () {
        if (mounted) {
          setState(() => _purchaseSuccess = false);
        }
      });
    }
  }

  /// Limpiar formulario
  void _clearForm() {
    _nameController.clear();
    _ageController.clear();
    _emailController.clear();
    _phoneController.clear();
    _cityController.clear();
  }

  /// Buscar juegos por nombre
  void _searchGames(String query) {
    if (query.isEmpty) {
      setState(() {});
    } else {
      setState(() {});
    }
  }

  /// Obtener juegos filtrados según búsqueda
  List<Game> _getFilteredGames() {
    final query = _searchController.text.toLowerCase();
    if (query.isEmpty) {
      return _allGames;
    }
    return _allGames
        .where((game) => game.name.toLowerCase().contains(query))
        .toList();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text(' Tienda de Videojuegos'),
        backgroundColor: AppConstants.primaryColor,
        elevation: 0,
        actions: [
          // Badge del carrito
          Padding(
            padding: const EdgeInsets.all(8.0),
            child: GestureDetector(
              onTap: () {
                setState(() => _showCart = !_showCart);
              },
              child: Stack(
                alignment: Alignment.topRight,
                children: [
                  const Icon(Icons.shopping_cart, size: 28),
                  if (_selectedGames.isNotEmpty)
                    Container(
                      padding: const EdgeInsets.all(2),
                      decoration: BoxDecoration(
                        color: Colors.red,
                        borderRadius: BorderRadius.circular(10),
                      ),
                      constraints: const BoxConstraints(
                        minWidth: 20,
                        minHeight: 20,
                      ),
                      child: Text(
                        _selectedGames.length.toString(),
                        style: const TextStyle(
                          color: Colors.white,
                          fontSize: 12,
                          fontWeight: FontWeight.bold,
                        ),
                        textAlign: TextAlign.center,
                      ),
                    ),
                ],
              ),
            ),
          ),
        ],
      ),
      body: Column(
        children: [
          // Barra de búsqueda
          Padding(
            padding: const EdgeInsets.all(16.0),
            child: TextField(
              controller: _searchController,
              onChanged: _searchGames,
              decoration: InputDecoration(
                hintText: 'Buscar juegos...',
                prefixIcon: const Icon(Icons.search),
                border: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(10),
                ),
                filled: true,
                fillColor: Colors.grey[100],
              ),
            ),
          ),

          // Contenido principal
          Expanded(
            child: _isLoading
                ? const Center(child: CircularProgressIndicator())
                : _showCheckout
                ? _buildCheckoutForm()
                : _showCart
                ? _buildCart()
                : _buildGamesList(),
          ),
        ],
      ),
      floatingActionButton: !_showCheckout && !_purchaseSuccess
          ? FloatingActionButton(
              onPressed: _selectedGames.isEmpty
                  ? null
                  : () {
                      setState(() => _showCheckout = true);
                    },
              backgroundColor: _selectedGames.isEmpty
                  ? Colors.grey
                  : AppConstants.accentColor,
              child: const Icon(Icons.shopping_bag),
            )
          : null,
    );
  }

  /// Widget de lista de juegos
  Widget _buildGamesList() {
    final filteredGames = _getFilteredGames();

    if (filteredGames.isEmpty) {
      return Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Icon(Icons.search_off, size: 64, color: Colors.grey[400]),
            const SizedBox(height: 16),
            Text(
              'No se encontraron juegos',
              style: TextStyle(fontSize: 18, color: Colors.grey[600]),
            ),
          ],
        ),
      );
    }

    return GridView.builder(
      padding: const EdgeInsets.all(12),
      gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
        crossAxisCount: 2,
        childAspectRatio: 0.6,
        crossAxisSpacing: 12,
        mainAxisSpacing: 12,
      ),
      itemCount: filteredGames.length,
      itemBuilder: (context, index) {
        final game = filteredGames[index];
        final isSelected = _selectedGames.any((g) => g.id == game.id);

        return GestureDetector(
          onTap: () => _toggleGameSelection(game),
          child: Card(
            elevation: isSelected ? 8 : 2,
            shape: RoundedRectangleBorder(
              borderRadius: BorderRadius.circular(12),
              side: isSelected
                  ? BorderSide(color: AppConstants.accentColor, width: 3)
                  : BorderSide.none,
            ),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                // Imagen del juego
                Expanded(
                  child: Container(
                    decoration: BoxDecoration(
                      borderRadius: const BorderRadius.only(
                        topLeft: Radius.circular(12),
                        topRight: Radius.circular(12),
                      ),
                      color: Colors.grey[300],
                    ),
                    child: game.imageUrl.isNotEmpty
                        ? Image.network(
                            game.imageUrl,
                            fit: BoxFit.cover,
                            errorBuilder: (context, error, stackTrace) {
                              return Container(
                                color: Colors.grey[400],
                                child: const Icon(Icons.image_not_supported),
                              );
                            },
                          )
                        : Container(
                            color: Colors.grey[400],
                            child: const Icon(Icons.image_not_supported),
                          ),
                  ),
                ),

                // Información del juego
                Padding(
                  padding: const EdgeInsets.all(8.0),
                  child: Column(
                    crossAxisAlignment: CrossAxisAlignment.start,
                    children: [
                      Text(
                        game.name,
                        maxLines: 2,
                        overflow: TextOverflow.ellipsis,
                        style: const TextStyle(
                          fontWeight: FontWeight.bold,
                          fontSize: 12,
                        ),
                      ),
                      const SizedBox(height: 4),
                      Text(
                        game.genres,
                        maxLines: 1,
                        overflow: TextOverflow.ellipsis,
                        style: TextStyle(fontSize: 10, color: Colors.grey[600]),
                      ),
                      const SizedBox(height: 8),
                      // Checkbox de selección
                      Container(
                        decoration: BoxDecoration(
                          color: isSelected
                              ? AppConstants.accentColor
                              : Colors.grey[200],
                          borderRadius: BorderRadius.circular(6),
                        ),
                        padding: const EdgeInsets.all(4),
                        child: Icon(
                          isSelected ? Icons.check : Icons.add_shopping_cart,
                          size: 16,
                          color: isSelected ? Colors.white : Colors.grey[600],
                        ),
                      ),
                    ],
                  ),
                ),
              ],
            ),
          ),
        );
      },
    );
  }

  /// Widget del carrito
  Widget _buildCart() {
    if (_selectedGames.isEmpty) {
      return Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Icon(
              Icons.shopping_cart_outlined,
              size: 64,
              color: Colors.grey[400],
            ),
            const SizedBox(height: 16),
            Text(
              'El carrito está vacío',
              style: TextStyle(fontSize: 18, color: Colors.grey[600]),
            ),
            const SizedBox(height: 24),
            ElevatedButton(
              onPressed: () {
                setState(() => _showCart = false);
              },
              style: ElevatedButton.styleFrom(
                backgroundColor: AppConstants.primaryColor,
                padding: const EdgeInsets.symmetric(
                  horizontal: 32,
                  vertical: 12,
                ),
              ),
              child: const Text('Continuar Comprando'),
            ),
          ],
        ),
      );
    }

    return Column(
      children: [
        Expanded(
          child: ListView.builder(
            padding: const EdgeInsets.all(12),
            itemCount: _selectedGames.length,
            itemBuilder: (context, index) {
              final game = _selectedGames[index];

              return Card(
                margin: const EdgeInsets.only(bottom: 12),
                child: ListTile(
                  leading: game.imageUrl.isNotEmpty
                      ? Image.network(
                          game.imageUrl,
                          width: 60,
                          height: 60,
                          fit: BoxFit.cover,
                          errorBuilder: (context, error, stackTrace) {
                            return const Icon(Icons.image_not_supported);
                          },
                        )
                      : const Icon(Icons.image_not_supported),
                  title: Text(game.name),
                  subtitle: Text(game.genres),
                  trailing: IconButton(
                    icon: const Icon(Icons.remove_circle),
                    onPressed: () => _toggleGameSelection(game),
                    color: Colors.red,
                  ),
                ),
              );
            },
          ),
        ),

        // Footer con botones
        Container(
          padding: const EdgeInsets.all(16),
          decoration: BoxDecoration(
            color: Colors.grey[100],
            borderRadius: const BorderRadius.only(
              topLeft: Radius.circular(12),
              topRight: Radius.circular(12),
            ),
          ),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.stretch,
            children: [
              Row(
                mainAxisAlignment: MainAxisAlignment.spaceBetween,
                children: [
                  Text(
                    'Total de juegos: ${_selectedGames.length}',
                    style: const TextStyle(
                      fontSize: 16,
                      fontWeight: FontWeight.bold,
                    ),
                  ),
                  Text(
                    'Total: \$${Purchase.calculateTotal(_selectedGames).toStringAsFixed(2)}',
                    style: const TextStyle(
                      fontSize: 16,
                      fontWeight: FontWeight.bold,
                      color: Colors.green,
                    ),
                  ),
                ],
              ),
              const SizedBox(height: 12),
              ElevatedButton(
                onPressed: () {
                  setState(() => _showCheckout = true);
                },
                style: ElevatedButton.styleFrom(
                  backgroundColor: AppConstants.accentColor,
                  padding: const EdgeInsets.symmetric(vertical: 12),
                ),
                child: const Text(
                  'Proceder al Pago',
                  style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold),
                ),
              ),
              const SizedBox(height: 8),
              TextButton(
                onPressed: () {
                  setState(() => _showCart = false);
                },
                child: const Text('Continuar Comprando'),
              ),
            ],
          ),
        ),
      ],
    );
  }

  /// Widget del formulario de compra
  Widget _buildCheckoutForm() {
    return SingleChildScrollView(
      child: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Form(
          key: _formKey,
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              // Título
              const Text(
                ' Datos de Compra',
                style: TextStyle(fontSize: 20, fontWeight: FontWeight.bold),
              ),
              const SizedBox(height: 12),

              // Campo: Nombre
              TextFormField(
                controller: _nameController,
                decoration: InputDecoration(
                  labelText: 'Nombre Completo *',
                  prefixIcon: const Icon(Icons.person),
                  border: OutlineInputBorder(
                    borderRadius: BorderRadius.circular(8),
                  ),
                ),
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return 'El nombre es requerido';
                  }
                  if (value.length < 3) {
                    return 'El nombre debe tener al menos 3 caracteres';
                  }
                  return null;
                },
              ),
              const SizedBox(height: 16),

              // Campo: Edad
              TextFormField(
                controller: _ageController,
                decoration: InputDecoration(
                  labelText: 'Edad *',
                  prefixIcon: const Icon(Icons.event_note),
                  border: OutlineInputBorder(
                    borderRadius: BorderRadius.circular(8),
                  ),
                ),
                keyboardType: TextInputType.number,
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return 'La edad es requerida';
                  }
                  try {
                    int age = int.parse(value);
                    if (age < 13 || age > 120) {
                      return 'La edad debe estar entre 13 y 120';
                    }
                  } catch (e) {
                    return 'Ingresa una edad válida';
                  }
                  return null;
                },
              ),
              const SizedBox(height: 16),

              // Campo: Email
              TextFormField(
                controller: _emailController,
                decoration: InputDecoration(
                  labelText: 'Correo Electrónico *',
                  prefixIcon: const Icon(Icons.email),
                  border: OutlineInputBorder(
                    borderRadius: BorderRadius.circular(8),
                  ),
                ),
                keyboardType: TextInputType.emailAddress,
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return 'El correo es requerido';
                  }
                  if (!value.contains('@')) {
                    return 'Ingresa un correo válido';
                  }
                  return null;
                },
              ),
              const SizedBox(height: 16),

              // Campo: Teléfono
              TextFormField(
                controller: _phoneController,
                decoration: InputDecoration(
                  labelText: 'Número de Teléfono *',
                  prefixIcon: const Icon(Icons.phone),
                  border: OutlineInputBorder(
                    borderRadius: BorderRadius.circular(8),
                  ),
                ),
                keyboardType: TextInputType.phone,
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return 'El teléfono es requerido';
                  }
                  if (value.length < 10) {
                    return 'Ingresa un teléfono válido';
                  }
                  return null;
                },
              ),
              const SizedBox(height: 16),

              // Campo: Ciudad
              TextFormField(
                controller: _cityController,
                decoration: InputDecoration(
                  labelText: 'Ciudad *',
                  prefixIcon: const Icon(Icons.location_city),
                  border: OutlineInputBorder(
                    borderRadius: BorderRadius.circular(8),
                  ),
                ),
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return 'La ciudad es requerida';
                  }
                  if (value.length < 2) {
                    return 'Ingresa una ciudad válida';
                  }
                  return null;
                },
              ),
              const SizedBox(height: 24),

              // Resumen de compra
              Container(
                padding: const EdgeInsets.all(12),
                decoration: BoxDecoration(
                  color: Colors.blue[50],
                  borderRadius: BorderRadius.circular(8),
                  border: Border.all(color: Colors.blue[300]!),
                ),
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    const Text(
                      'Resumen de Compra:',
                      style: TextStyle(
                        fontWeight: FontWeight.bold,
                        fontSize: 14,
                      ),
                    ),
                    const SizedBox(height: 8),
                    Text('Juegos seleccionados: ${_selectedGames.length}'),
                    Text(
                      'Total: \$${Purchase.calculateTotal(_selectedGames).toStringAsFixed(2)}',
                      style: const TextStyle(
                        fontWeight: FontWeight.bold,
                        color: Colors.green,
                      ),
                    ),
                  ],
                ),
              ),
              const SizedBox(height: 24),

              // Botones
              SizedBox(
                width: double.infinity,
                child: Column(
                  children: [
                    ElevatedButton(
                      onPressed: _processPurchase,
                      style: ElevatedButton.styleFrom(
                        backgroundColor: Colors.green,
                        padding: const EdgeInsets.symmetric(vertical: 16),
                      ),
                      child: const Text(
                        ' PAGAR',
                        style: TextStyle(
                          fontSize: 18,
                          fontWeight: FontWeight.bold,
                          color: Colors.white,
                        ),
                      ),
                    ),
                    const SizedBox(height: 12),
                    TextButton(
                      onPressed: () {
                        setState(() => _showCheckout = false);
                      },
                      child: const Text('Volver'),
                    ),
                  ],
                ),
              ),
              const SizedBox(height: 24),
            ],
          ),
        ),
      ),
    );
  }
}
