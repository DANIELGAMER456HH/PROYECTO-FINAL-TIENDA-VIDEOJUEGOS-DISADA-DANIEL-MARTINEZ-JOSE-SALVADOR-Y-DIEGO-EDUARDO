  @override
  void initState() {
    super.initState();

    _searchController = TextEditingController();
    _searchController.addListener(_onSearchChanged);

    _initializeStorage();
    _loadGames();
  }

  /// Inicializa el almacenamiento
  Future<void> _initializeStorage() async {
    try {
      await _gameStorage.init();

      // Cargar IDs de favoritos
      final favorites = await _gameStorage.getFavoritesGames();
      setState(() {
        _favoriteIds = favorites.map((g) => g.id).toList();
      });
    } catch (e) {
      print('❌ Error inicializando almacenamiento: $e');
    }
  }

  /// Carga los juegos iniciales
  Future<void> _loadGames() async {
    if (_isLoading) return;

    setState(() {
      _isLoading = true;
    });

    try {
      final games = await _apiService.getPopularGames(
        page: _currentPage,
        pageSize: 20,
      );

      setState(() {
        if (_currentPage == 1) {
          _games = games;
        } else {
          _games.addAll(games);
        }

        _hasMoreGames = games.length == 20;
        _isLoading = false;
      });

      // Guardar en historial
      for (var game in games.take(5)) {
        await _gameStorage.addToHistory(game);
      }
    } catch (e) {
      setState(() {
        _isLoading = false;
      });

      _showErrorSnackBar('Error cargando juegos: $e');
    }
  }

  /// Carga más juegos (paginación)
  void _loadMoreGames() {
    if (_isLoadingMore || !_hasMoreGames) return;

    setState(() {
      _isLoadingMore = true;
      _currentPage++;
    });

    _loadGames();

    setState(() {
      _isLoadingMore = false;
    });
  }

  /// Manejador de cambios en la búsqueda
  void _onSearchChanged() {
    final query = _searchController.text.trim();

    if (query.isEmpty) {
      setState(() {
        _isSearching = false;
        _searchResults = [];
        _currentPage = 1;
      });
      _loadGames();
      return;
    }

    setState(() {
      _isSearching = true;
    });

    _performSearch(query);
  }

  /// Realiza la búsqueda
  Future<void> _performSearch(String query) async {
    try {
      final results = await _apiService.searchGames(query);

      setState(() {
        _searchResults = results;
      });
    } catch (e) {
      _showErrorSnackBar('Error en la búsqueda: $e');
    }
  }

  /// Maneja el cambio de favorito
  Future<void> _toggleFavorite(Game game, bool isFavorite) async {
    try {
      if (isFavorite) {
        await _gameStorage.addFavorite(game);
        setState(() {
          _favoriteIds.add(game.id);
        });
        _showSuccessSnackBar('${game.name} agregado a favoritos ⭐');
      } else {
        await _gameStorage.removeFavorite(game.id);
        setState(() {
          _favoriteIds.remove(game.id);
        });
        _showSuccessSnackBar('${game.name} removido de favoritos');
      }
    } catch (e) {
      _showErrorSnackBar('Error: $e');
    }
  }

  /// Muestra un mensaje de éxito
  void _showSuccessSnackBar(String message) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        content: Text(message),
        backgroundColor: Colors.green[700],
        duration: const Duration(seconds: 2),
      ),
    );
  }

  /// Muestra un mensaje de error
  void _showErrorSnackBar(String message) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        content: Text(message),
        backgroundColor: Colors.red[700],
        duration: const Duration(seconds: 3),
      ),
    );
  }

  @override
  void dispose() {
    _searchController.dispose();
    _apiService.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final gamesToShow = _isSearching ? _searchResults : _games;

    return Scaffold(
      appBar: AppBar(
        title: const Text('🎮 Explorar Juegos'),
        elevation: 0,
        actions: [
          if (!_isSearching)
            Padding(
              padding: const EdgeInsets.all(8.0),
              child: Tooltip(
                message: 'Actualizar',
                child: IconButton(
                  icon: const Icon(Icons.refresh),
                  onPressed: () {
                    _currentPage = 1;
                    _loadGames();
                  },
                ),
              ),
            ),
        ],
      ),
      body: Column(
        children: [
          // ========== BÚSQUEDA ==========
          Padding(
            padding: const EdgeInsets.all(12),
            child: TextField(
              controller: _searchController,
              cursorColor: const Color(0xFF00D9FF),
              style: const TextStyle(color: Colors.white),
              decoration: InputDecoration(
                hintText: 'Buscar juegos...',
                hintStyle: TextStyle(color: Colors.grey[500]),
                prefixIcon: const Icon(Icons.search, color: Color(0xFF00D9FF)),
                suffixIcon: _searchController.text.isNotEmpty
                    ? IconButton(
                        icon: const Icon(Icons.clear, color: Color(0xFF00D9FF)),
                        onPressed: () {
                          _searchController.clear();
                        },
                      )
                    : null,
                filled: true,
                fillColor: Colors.grey[900],
                border: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(12),
                  borderSide: const BorderSide(color: Color(0xFF00D9FF)),
                ),
                focusedBorder: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(12),
                  borderSide: const BorderSide(
                    color: Color(0xFF00D9FF),
                    width: 2,
                  ),
                ),
              ),
            ),
          ),

          // ========== LISTA DE JUEGOS ==========
          Expanded(
            child: _isLoading && gamesToShow.isEmpty
                ? const LoadingWidget(message: 'Cargando juegos...')
                : gamesToShow.isEmpty
                ? EmptyStateWidget(
                    title: _isSearching
                        ? 'Sin resultados'
                        : 'Sin juegos disponibles',
                    message: _isSearching
                        ? 'No encontramos juegos que coincidan'
                        : 'No hay juegos para mostrar',
                    icon: _isSearching
                        ? Icons.search_off
                        : Icons.inbox_outlined,
                  )
                : GridView.builder(
                    padding: const EdgeInsets.symmetric(horizontal: 12),
                    gridDelegate:
                        const SliverGridDelegateWithFixedCrossAxisCount(
                          crossAxisCount: 2,
                          childAspectRatio: 0.6,
                          crossAxisSpacing: 12,
                          mainAxisSpacing: 12,
                        ),
                    itemCount:
                        gamesToShow.length +
                        (_hasMoreGames && !_isSearching ? 1 : 0),
                    itemBuilder: (context, index) {
                      // Widget de carga en la última posición
                      if (index == gamesToShow.length) {
                        _loadMoreGames();
                        return const LoadingWidget(message: 'Cargando más...');
                      }

                      final game = gamesToShow[index];
                      final isFavorite = _favoriteIds.contains(game.id);

                      return GameCard(
                        game: game.copyWith(isFavorite: isFavorite),
                        onTap: () {
                          Navigator.pushNamed(
                            context,
                            '/game-detail',
                            arguments: game.id,
                          );
                        },
                        onFavoriteToggle: (newState) {
                          _toggleFavorite(game, newState);
                        },
                      );
                    },
                  ),
          ),
        ],
      ),
    );
  }
